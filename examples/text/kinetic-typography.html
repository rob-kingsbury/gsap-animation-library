// Split text into characters
function splitChars(element) {
    const text = element.textContent;
    element.innerHTML = text.split('')
        .map((char, i) =>
            char === ' ' ? ' ' :
            `<span class="char" style="--char-index:${i}">${char}</span>`
        ).join('');
    return element.querySelectorAll('.char');
}
// Wave animation with stagger
const chars = splitChars(element);

gsap.to(chars, {
    y: -30,
    duration: 0.4,
    ease: "sine.inOut",
    stagger: {
        each: 0.1,
        repeat: -1,
        yoyo: true
    }
});
// 3D perspective rotation
gsap.from(chars, {
    rotationX: -90,
    rotationY: 45,
    opacity: 0,
    transformOrigin: "50% 50% -50px",
    stagger: 0.1,
    duration: 1,
    ease: "back.out(1.7)"
});
// Magnetic effect on mousemove
element.addEventListener('mousemove', (e) => {
    const rect = element.getBoundingClientRect();
    const centerX = rect.left + rect.width / 2;
    const centerY = rect.top + rect.height / 2;

    chars.forEach(char => {
        const charRect = char.getBoundingClientRect();
        const charCenterX = charRect.left + charRect.width / 2;
        const charCenterY = charRect.top + charRect.height / 2;

        const deltaX = (e.clientX - charCenterX) * 0.1;
        const deltaY = (e.clientY - charCenterY) * 0.1;

        gsap.to(char, {
            x: deltaX,
            y: deltaY,
            duration: 0.3
        });
    });
});
