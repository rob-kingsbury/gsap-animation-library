// Basic Typewriter with GSAP
function typewriter(element, text, options = {}) {
    const {
        speed = 0.05,     // seconds per character
        delay = 0,        // initial delay
        onComplete = null
    } = options;

    const chars = text.split('');
    element.textContent = '';

    return gsap.to({}, {
        duration: chars.length * speed,
        delay: delay,
        ease: "none",
        onUpdate: function() {
            const progress = this.progress();
            const charIndex = Math.floor(progress * chars.length);
            element.textContent = text.substring(0, charIndex);
        },
        onComplete: () => {
            element.textContent = text;
            if (onComplete) onComplete();
        }
    });
}
// Delete text effect
function deleteText(element, options = {}) {
    const { speed = 0.03, delay = 0, onComplete } = options;
    const text = element.textContent;

    return gsap.to({}, {
        duration: text.length * speed,
        delay: delay,
        ease: "none",
        onUpdate: function() {
            const progress = this.progress();
            const charsToShow = text.length - Math.floor(progress * text.length);
            element.textContent = text.substring(0, charsToShow);
        },
        onComplete: () => {
            element.textContent = '';
            if (onComplete) onComplete();
        }
    });
}
// Rotating words typewriter
const words = ['websites', 'apps', 'experiences'];
let wordIndex = 0;

function rotateWords() {
    const word = words[wordIndex];

    // Type the word
    typewriter(element, word, {
        speed: 0.08,
        onComplete: () => {
            // Wait, then delete
            gsap.delayedCall(2, () => {
                deleteText(element, {
                    speed: 0.04,
                    onComplete: () => {
                        wordIndex = (wordIndex + 1) % words.length;
                        rotateWords();
                    }
                });
            });
        }
    });
}

rotateWords();
