Code Reference

// Basic bento grid stagger reveal
gsap.to(".bento-item", {
    opacity: 1,
    y: 0,
    scale: 1,
    duration: 0.6,
    stagger: {
        amount: 0.8,
        grid: "auto",
        from: "start"  // or "center", "edges", "random"
    },
    ease: "power3.out",
    scrollTrigger: {
        trigger: ".bento-grid",
        start: "top 70%"
    }
});
// Alternating slide directions
gsap.to(".bento-slide-left .bento-item", {
    opacity: 1,
    x: 0,
    duration: 0.6,
    stagger: 0.1,
    ease: "power3.out",
    scrollTrigger: {
        trigger: ".bento-slide-left",
        start: "top 70%"
    }
});
// Rotation reveal
gsap.to(".bento-rotate .bento-item", {
    opacity: 1,
    y: 0,
    rotation: 0,
    duration: 0.7,
    stagger: {
        amount: 0.6,
        from: "random"
    },
    ease: "back.out(1.2)",
    scrollTrigger: {
        trigger: ".bento-rotate",
        start: "top 70%"
    }
});
// CSS Grid for Bento Layout
.bento-grid {
    display: grid;
    gap: 1.5rem;
}

.bento-classic {
    grid-template-columns: repeat(4, 1fr);
    grid-template-rows: repeat(2, 300px);
}

/* First item spans 2x2 */
.bento-classic .bento-item:nth-child(1) {
    grid-column: span 2;
    grid-row: span 2;
}
