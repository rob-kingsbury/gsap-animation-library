// CSS-based flip (simplest approach)
// HTML: Toggle class on click
card.addEventListener('click', () => {
    card.classList.toggle('is-flipped');
});

/* CSS */
.flip-card {
    perspective: 1000px;
}

.flip-card-inner {
    transform-style: preserve-3d;
    transition: transform 0.8s ease;
}

.flip-card.is-flipped .flip-card-inner {
    transform: rotateY(180deg);
}

.flip-card-front, .flip-card-back {
    backface-visibility: hidden;
}

.flip-card-back {
    transform: rotateY(180deg);
}
// GSAP-controlled flip
function flipCard(card, options = {}) {
    const {
        duration = 0.8,
        ease = "power2.inOut",
        direction = "horizontal"
    } = options;

    const inner = card.querySelector('.flip-card-inner');
    const isFlipped = card.classList.contains('is-flipped');
    const rotation = direction === 'horizontal' ? 'rotateY' : 'rotateX';

    gsap.to(inner, {
        [rotation]: isFlipped ? 0 : 180,
        duration,
        ease,
        onComplete: () => {
            card.classList.toggle('is-flipped');
        }
    });
}
// Scroll-triggered reveal with stagger
gsap.to(".reveal-card", {
    rotateY: 0,
    opacity: 1,
    duration: 0.8,
    stagger: 0.15,
    ease: "back.out(1.4)",
    scrollTrigger: {
        trigger: ".card-grid",
        start: "top 70%"
    }
});
// Bounce flip effect
function bounceFlip(card) {
    const inner = card.querySelector('.flip-card-inner');

    const tl = gsap.timeline();

    tl.to(inner, {
        scale: 0.9,
        duration: 0.2
    })
    .to(inner, {
        rotateY: 180,
        duration: 0.6,
        ease: "power2.inOut"
    })
    .to(inner, {
        scale: 1,
        duration: 0.3,
        ease: "back.out(1.7)"
    }, "-=0.2");
}
