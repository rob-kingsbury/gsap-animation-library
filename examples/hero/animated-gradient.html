Code Reference

// 1. Moving gradient background
gsap.to('.gradient-move', {
    backgroundPosition: '100% 100%',
    duration: 10,
    repeat: -1,
    yoyo: true,
    ease: "sine.inOut"
});

/* CSS */
.gradient-move {
    background: linear-gradient(-45deg, #color1, #color2, #color3, #color4);
    background-size: 400% 400%;
}
// 2. Radial pulse
gsap.to('.radial-pulse', {
    opacity: 1,
    scale: 1.5,
    duration: 2,
    repeat: -1,
    yoyo: true,
    ease: "sine.inOut"
});
// 3. Gradient text shimmer
gsap.to('.gradient-text', {
    backgroundPosition: '-200% 0',
    duration: 3,
    repeat: -1,
    ease: "none"
});

/* CSS */
.gradient-text {
    background: linear-gradient(90deg, #gold, #light, #gold, #dark, #gold);
    background-size: 200% 100%;
    -webkit-background-clip: text;
    -webkit-text-fill-color: transparent;
}
// 4. Mesh gradient blobs
gsap.to('.mesh-blob-1', {
    x: 100,
    y: 50,
    scale: 1.2,
    duration: 8,
    repeat: -1,
    yoyo: true,
    ease: "sine.inOut"
});

/* CSS */
.mesh-blob {
    filter: blur(80px);
    border-radius: 50%;
}
// 5. Mouse-following spotlight
document.addEventListener('mousemove', (e) => {
    gsap.to('.spotlight', {
        left: e.clientX,
        top: e.clientY,
        duration: 0.3,
        ease: "power2.out"
    });
});
