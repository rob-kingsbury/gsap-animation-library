Slide-Out Drawers

Off-canvas panels that slide in from any direction with smooth GSAP animations

Drawer Features

Multiple Directions

Drawers can slide in from left, right, top, or bottom depending on your needs.

Staggered Content

Menu items animate in with staggered timing for a polished reveal effect.

Push Effect

Main content can optionally push aside when drawers open (left/right).

Overlay Dismiss

Click the overlay or press Escape to close the drawer quickly.

Code Reference

// Basic drawer animation
function openDrawer(drawer, direction) {
    const tl = gsap.timeline();

    // Determine transform based on direction
    const transforms = {
        left: { x: 0 },
        right: { x: 0 },
        top: { y: 0 },
        bottom: { y: 0 }
    };

    tl.to(drawer, {
        ...transforms[direction],
        duration: 0.5,
        ease: "power3.out"
    })
    .to(drawer.querySelectorAll('.drawer-nav a'), {
        opacity: 1,
        x: 0,
        y: 0,
        stagger: 0.05,
        duration: 0.3
    }, "-=0.2");

    return tl;
}
// Push main content effect
if (direction === 'left') {
    gsap.to('.main-content', {
        x: drawerWidth,
        duration: 0.5,
        ease: "power3.out"
    });
}

// Overlay fade in
gsap.to('.overlay', {
    opacity: 1,
    visibility: "visible",
    duration: 0.3
});
// Close drawer with reverse animation
function closeDrawer(drawer, direction) {
    const transforms = {
        left: { x: '-100%' },
        right: { x: '100%' },
        top: { y: '-100%' },
        bottom: { y: '100%' }
    };

    const tl = gsap.timeline();

    tl.to(drawer.querySelectorAll('.drawer-nav a'), {
        opacity: 0,
        duration: 0.2
    })
    .to(drawer, {
        ...transforms[direction],
        duration: 0.4,
        ease: "power3.in"
    }, "-=0.1");

    return tl;
}
