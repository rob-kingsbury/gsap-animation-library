Code Reference

// Animate indicator to active tab
function moveIndicator(indicator, activeBtn) {
    gsap.to(indicator, {
        left: activeBtn.offsetLeft,
        width: activeBtn.offsetWidth,
        duration: 0.4,
        ease: "power3.out"
    });
}
// Content transition - Fade
function transitionFade(oldPanel, newPanel) {
    const tl = gsap.timeline();

    tl.to(oldPanel, {
        opacity: 0,
        duration: 0.2,
        onComplete: () => {
            oldPanel.classList.remove('is-active');
        }
    })
    .fromTo(newPanel,
        { opacity: 0 },
        {
            opacity: 1,
            duration: 0.3,
            onStart: () => {
                newPanel.classList.add('is-active');
            }
        }
    );
}
// Content transition - Slide with direction
function transitionSlide(oldPanel, newPanel, direction) {
    const xOffset = direction > 0 ? 50 : -50;

    const tl = gsap.timeline();

    tl.to(oldPanel, {
        x: -xOffset,
        opacity: 0,
        duration: 0.25,
        onComplete: () => {
            oldPanel.classList.remove('is-active');
            gsap.set(oldPanel, { x: 0 });
        }
    })
    .fromTo(newPanel,
        { x: xOffset, opacity: 0 },
        {
            x: 0,
            opacity: 1,
            duration: 0.3,
            onStart: () => {
                newPanel.classList.add('is-active');
            }
        }
    );
}
