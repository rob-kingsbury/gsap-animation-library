Code Reference

// Basic reveal pattern with ScrollTrigger
gsap.from(".reveal-element", {
    opacity: 0,
    y: 60,
    duration: 1,
    ease: "power3.out",
    scrollTrigger: {
        trigger: ".reveal-element",
        start: "top 80%",   // When top hits 80% of viewport
        toggleActions: "play none none none"
    }
});
// Batch reveals for better performance
ScrollTrigger.batch(".reveal-fade-up", {
    start: "top 85%",
    onEnter: (elements) => {
        gsap.to(elements, {
            opacity: 1,
            y: 0,
            stagger: 0.1,
            duration: 0.8
        });
    }
});
// Clip-path reveal
gsap.to(".reveal-clip", {
    clipPath: "inset(0 0 0 0)",
    duration: 1.2,
    ease: "power4.inOut",
    scrollTrigger: {
        trigger: ".reveal-clip",
        start: "top 75%"
    }
});
// Stagger grid reveal
gsap.to(".stagger-item", {
    opacity: 1,
    scale: 1,
    duration: 0.5,
    stagger: {
        amount: 0.8,      // Total time for all staggers
        grid: "auto",      // Auto-detect grid
        from: "center"    // Start from center
    },
    ease: "back.out(1.7)",
    scrollTrigger: {
        trigger: ".stagger-grid",
        start: "top 70%"
    }
});
