// Key Pattern: Horizontal Scroll with ScrollTrigger

// 1. Calculate total scroll distance
const wrapper = document.querySelector('.horizontal-wrapper');
const scrollDistance = wrapper.scrollWidth - window.innerWidth;

// 2. Create the horizontal scroll animation
gsap.to(wrapper, {
    x: -scrollDistance,       // Move left by scroll distance
    ease: "none",              // Linear movement for scroll
    scrollTrigger: {
        trigger: ".horizontal-section",
        pin: true,              // Pin section during scroll
        scrub: 1,               // Smooth scrubbing (1 second)
        end: `+=${scrollDistance}`,  // Scroll distance
    }
});

// 3. Add progress indicator
gsap.to(".progress-bar-fill", {
    width: "100%",
    ease: "none",
    scrollTrigger: {
        trigger: ".horizontal-section",
        scrub: true,
        start: "top top",
        end: `+=${scrollDistance}`,
    }
});
