Code Reference

// Clip-path slide reveal
gsap.to(".slide-card.from-left .slide-card-content", {
    clipPath: "inset(0 0 0 0)",
    duration: 1,
    ease: "power4.inOut",
    scrollTrigger: {
        trigger: ".slide-card",
        start: "top 70%"
    }
});
// Mask overlay reveal
const tl = gsap.timeline({
    scrollTrigger: {
        trigger: ".gallery-slide-item",
        start: "top 75%"
    }
});

// Slide mask away
tl.to(".gallery-slide-mask", {
    scaleX: 0,
    transformOrigin: "right center",
    duration: 0.8,
    ease: "power4.inOut"
});

// Scale image to normal
tl.to("img", {
    scale: 1,
    duration: 1,
    ease: "power3.out"
}, 0);
// Text slide from left
gsap.to(".text-slide-inner", {
    x: 0,
    duration: 0.8,
    stagger: 0.15,
    ease: "power4.out",
    scrollTrigger: {
        trigger: ".text-slide-container",
        start: "top 70%"
    }
});
// Alternating direction slides
gsap.to(".stacked-card", {
    x: 0,
    opacity: 1,
    duration: 0.6,
    stagger: 0.15,
    ease: "power3.out",
    scrollTrigger: {
        trigger: ".stacked-slide",
        start: "top 70%"
    }
});
