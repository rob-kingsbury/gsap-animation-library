// Coordinated split section animation
function animateSplitSection(section) {
    const content = section.querySelector('.split-content');
    const image = section.querySelector('.split-image');
    const glow = section.querySelector('.bg-glow');

    const tl = gsap.timeline({
        scrollTrigger: {
            trigger: section,
            start: 'top 70%',
            end: 'top 20%',
            toggleActions: 'play none none none'
        }
    });

    // Background glow fades in
    tl.to(glow, { opacity: 1, duration: 1 }, 0);

    // Content slides in
    tl.to(content, {
        opacity: 1,
        x: 0,
        duration: 0.8,
        ease: 'power3.out'
    }, 0.2);

    // Image slides in (slightly delayed)
    tl.to(image, {
        opacity: 1,
        x: 0,
        duration: 0.8,
        ease: 'power3.out'
    }, 0.4);
}

// Number counter animation
function animateCounter(element) {
    const target = +element.dataset.target;

    gsap.to(element, {
        innerText: target,
        duration: 2,
        ease: 'power2.out',
        snap: { innerText: 1 },
        scrollTrigger: {
            trigger: element,
            start: 'top 80%'
        }
    });
}
