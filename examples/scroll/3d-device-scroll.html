Code Reference

// 3D Device scroll animation with ScrollTrigger
gsap.to(".device", {
    rotateY: 0,
    rotateX: 0,
    translateZ: 0,
    opacity: 1,
    duration: 1,
    ease: "power3.out",
    scrollTrigger: {
        trigger: ".device-section",
        start: "top 60%",
        end: "top 20%",
        scrub: 1  // Smooth scrubbing
    }
});
// Laptop lid opening effect
gsap.to(".laptop-screen", {
    rotateX: 0,
    ease: "power2.out",
    scrollTrigger: {
        trigger: ".laptop-wrapper",
        start: "top 70%",
        end: "top 30%",
        scrub: 1.5
    }
});
// Multiple devices with stagger
gsap.to(".multi-device", {
    rotateY: 0,
    rotateX: 0,
    opacity: 1,
    stagger: 0.2,
    duration: 1,
    ease: "power3.out",
    scrollTrigger: {
        trigger: ".multi-devices",
        start: "top 70%"
    }
});
// CSS for 3D perspective
.device-wrapper {
    perspective: 1500px;
}

.device {
    transform-style: preserve-3d;
    /* Initial rotated state */
    transform: rotateY(-35deg) rotateX(15deg) translateZ(-100px);
    opacity: 0;
}
