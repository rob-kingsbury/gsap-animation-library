Code Reference

// Circle Spinner
gsap.to('.spinner-circle', {
    rotation: 360,
    duration: 1,
    repeat: -1,
    ease: "none"
});
// Dots Wave
gsap.to('.dot', {
    y: -15,
    duration: 0.4,
    repeat: -1,
    yoyo: true,
    stagger: {
        each: 0.15,
        repeat: -1,
        yoyo: true
    },
    ease: "power2.inOut"
});
// Equalizer Bars
gsap.to('.bar', {
    scaleY: gsap.utils.random(0.3, 2, true),
    duration: 0.3,
    repeat: -1,
    yoyo: true,
    stagger: 0.1,
    ease: "power1.inOut"
});
// Pulse Effect
gsap.to('.spinner-pulse', {
    scale: 1.3,
    opacity: 0,
    duration: 1,
    repeat: -1,
    ease: "power2.out"
});
// Orbit Animation
gsap.to('.spinner-orbit', {
    rotation: 360,
    duration: 1.5,
    repeat: -1,
    ease: "none"
});
