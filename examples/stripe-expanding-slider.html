/* Key CSS - Pure CSS, no JS required! */

/* Container uses flexbox */
.stripe-slider {
  display: flex;
  flex-direction: column; /* or row for horizontal */
  gap: 8px;
}

/* Each slide starts with flex: 1 */
.stripe-slide {
  flex: 1;
  min-height: 60px; /* Collapsed height */
  transition: flex 0.5s cubic-bezier(0.4, 0, 0.2, 1);
}

/* On hover, expand to take 4x space */
.stripe-slide:hover {
  flex: 4;
}

/* Content fades/slides in on hover */
.stripe-slide-content {
  opacity: 0;
  transform: translateY(20px);
  transition: all 0.4s ease 0.1s;
}

.stripe-slide:hover .stripe-slide-content {
  opacity: 1;
  transform: translateY(0);
}
